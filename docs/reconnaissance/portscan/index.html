      Automated Bash Script link

      This script utilizes rustscan for faster scanning.

        #!/bin/bash

if [ "$#" -ne 1 ]; then
    echo "Usage: ports HOST"
    exit  2
fi

nmap -p- --min-rate 1000 $1 -oN all_ports -vvv -Pn
ports=$(grep -iv not all_ports | grep -i 'open\|filtered' | awk -F'/' '{ print $1 }' ORS=',')
rustscan --ulimit 5000 -a $1 -p $ports -- -Pn -A --script vuln -oN nmap
rm ./all_ports
